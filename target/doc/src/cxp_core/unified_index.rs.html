1//! Unified Cross-Modal Search Index
2//!
3//! Combines text and image embeddings in a single HNSW index for cross-modal retrieval.
4//! Uses SigLIP 2's shared 512-dimensional embedding space to enable:
5//! - Text-to-image search
6//! - Image-to-text search
7//! - Unified multimodal search
8//!
9//! Features:
10//! - Single unified index for all modalities
11//! - Metadata tracking for entry types
12//! - Type-specific filtering
13//! - Persistent storage support
14

30/// Type of entry in the unified index
31#[cfg(all(feature = "search", feature = "multimodal"))]
32#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
33pub enum EntryType {
34    /// Text chunk entry
35    Text {
36        /// Unique chunk ID
37        chunk_id: u64,
38        /// Source file path
39        file_path: String,
40    },
41    /// Image entry
42    Image {
43        /// Image file path
44        file_path: String,
45    },
46}
47
48impl EntryType {
49    /// Check if this is a text entry
50    pub fn is_text(&self) -> bool {
51        matches!(self, EntryType::Text { .. })
52    }
53
54    /// Check if this is an image entry
55    pub fn is_image(&self) -> bool {
56        matches!(self, EntryType::Image { .. })
57    }
58
59    /// Get the file path for this entry
60    pub fn file_path(&self) -> &str {
61        match self {
62            EntryType::Text { file_path, .. } => file_path,
63            EntryType::Image { file_path } => file_path,
64        }
65    }
66}

68/// Search result with type information
69#[cfg(all(feature = "search", feature = "multimodal"))]
70#[derive(Debug, Clone, PartialEq)]
71pub struct SearchResultWithType {
72    /// Vector ID
73    pub id: u64,
74    /// Distance to query (lower is more similar)
75    pub distance: f32,
76    /// Entry type and metadata
77    pub entry_type: EntryType,
78}
79
80impl SearchResultWithType {
81    /// Convert distance to similarity score (0.0 to 1.0)
82    pub fn similarity(&self) -> f32 {
83        // For cosine distance: similarity = 1 - distance
84        1.0 - self.distance
85    }
86}

88/// Unified index for cross-modal search
89///
90/// Stores both text and image embeddings in the same vector space,
91/// enabling semantic search across modalities.
92///
93/// # Example
94/// ```no_run
95/// use cxp_core::unified_index::UnifiedIndex;
96/// use cxp_core::index::HnswConfig;
97///
98/// # fn main() -> Result<(), Box<dyn std::error::Error>> {
99/// // Create a new unified index with 512-dim float32 embeddings
100/// let config = HnswConfig::multimodal_float32();
101/// let mut index = UnifiedIndex::new(config)?;
102///
103/// // Add text embedding
104/// let text_embedding = vec![0.1; 512]; // Placeholder
105/// index.add_text(1, &text_embedding, 42, "document.txt")?;
106///
107/// // Add image embedding
108/// let image_embedding = vec![0.2; 512]; // Placeholder
109/// index.add_image(2, &image_embedding, "photo.jpg")?;
110///
111/// // Search with text, find both images and text
112/// let results = index.search(&text_embedding, 10)?;
113///
114/// // Search only images with text query
115/// let image_results = index.search_images_only(&text_embedding, 5)?;
116/// # Ok(())
117/// # }
118/// ```
119#[cfg(all(feature = "search", feature = "multimodal"))]
120pub struct UnifiedIndex {
121    /// Underlying HNSW index
122    hnsw: HnswIndex,
123    /// Metadata for each entry
124    metadata: HashMap<u64, EntryType>,
125}
126
127#[cfg(all(feature = "search", feature = "multimodal"))]
128impl UnifiedIndex {
129    /// Create a new unified index with the given configuration
130    ///
131    /// Recommended configs:
132    /// - `HnswConfig::multimodal_float32()` for best accuracy (512-dim float32)
133    /// - `HnswConfig::multimodal_binary()` for best compression (64 bytes)
134    pub fn new(config: HnswConfig) -> Result<Self> {
135        let hnsw = HnswIndex::new(config)?;
136        let metadata = HashMap::new();
137
138        Ok(Self { hnsw, metadata })
139    }
140
141    /// Add a text embedding to the index
142    ///
143    /// # Arguments
144    /// * `id` - Unique ID for this vector (must be unique across all entries)
145    /// * `embedding` - 512-dimensional embedding vector
146    /// * `chunk_id` - Text chunk identifier
147    /// * `file_path` - Source file path
148    pub fn add_text(
149        &mut self,
150        id: u64,
151        embedding: &[f32],
152        chunk_id: u64,
153        file_path: &str,
154    ) -> Result<()> {
155        self.hnsw.add_f32(id, embedding)?;
156        self.metadata.insert(
157            id,
158            EntryType::Text {
159                chunk_id,
160                file_path: file_path.to_string(),
161            },
162        );
163        Ok(())
164    }
165
166    /// Add a text embedding using binary quantization
167    ///
168    /// More memory-efficient than float32, with minimal accuracy loss.
169    pub fn add_text_binary(
170        &mut self,
171        id: u64,
172        bits: &[u8],
173        chunk_id: u64,
174        file_path: &str,
175    ) -> Result<()> {
176        self.hnsw.add_binary(id, bits)?;
177        self.metadata.insert(
178            id,
179            EntryType::Text {
180                chunk_id,
181                file_path: file_path.to_string(),
182            },
183        );
184        Ok(())
185    }

218    /// Cross-modal search: find both text and images
219    ///
220    /// Returns all matching entries regardless of type, sorted by similarity.
221    ///
222    /// # Arguments
223    /// * `query` - Query embedding (can be from text or image)
224    /// * `top_k` - Number of results to return
225    pub fn search(&self, query: &[f32], top_k: usize) -> Result<Vec<SearchResultWithType>> {
226        let results = self.hnsw.search_f32(query, top_k)?;
227        self.attach_metadata(results)
228    }
229
230    /// Cross-modal search with binary query
231    pub fn search_binary(&self, bits: &[u8], top_k: usize) -> Result<Vec<SearchResultWithType>> {
232        let results = self.hnsw.search_binary(bits, top_k)?;
233        self.attach_metadata(results)
234    }
235
236    /// Search only for images
237    ///
238    /// Filters results to return only image entries. Useful for text-to-image search.
239    pub fn search_images_only(
240        &self,
241        query: &[f32],
242        top_k: usize,
243    ) -> Result<Vec<SearchResultWithType>> {
244        // Search more than requested to account for filtering
245        let search_k = top_k * 3; // Heuristic: search 3x to get enough images
246        let all_results = self.hnsw.search_f32(query, search_k)?;
247        let results_with_metadata = self.attach_metadata(all_results)?;
248
249        // Filter to images only
250        let filtered: Vec<_> = results_with_metadata
251            .into_iter()
252            .filter(|r| r.entry_type.is_image())
253            .take(top_k)
254            .collect();
255
256        Ok(filtered)
257    }

259    /// Search only for text
260    ///
261    /// Filters results to return only text entries. Useful for image-to-text search.
262    pub fn search_text_only(
263        &self,
264        query: &[f32],
265        top_k: usize,
266    ) -> Result<Vec<SearchResultWithType>> {
267        // Search more than requested to account for filtering

269        let all_results = self.hnsw.search_f32(query, search_k)?;

271
272        // Filter to text only

274            .into_iter()
275            .filter(|r| r.entry_type.is_text())
276            .take(top_k)
277            .collect();

279        Ok(filtered)
280    }

331    /// Save index and metadata to disk
332    ///
333    /// Saves two files:
334    /// - `{path}.index` - HNSW index data
335    /// - `{path}.meta` - Metadata (JSON)
336    pub fn save<P: AsRef<Path>>(&self, path: P) -> Result<()> {
337        let base_path = path.as_ref();
338
339        // Save HNSW index
340        let index_path = base_path.with_extension("index");
341        self.hnsw.save(&index_path)?;
342
343        // Save metadata as JSON
344        let meta_path = base_path.with_extension("meta");
345        let meta_json = serde_json::to_string(&self.metadata)
346            .map_err(|e| CxpError::Search(format!("Failed to serialize metadata: {}", e)))?;
347        std::fs::write(&meta_path, meta_json)
348            .map_err(|e| CxpError::Search(format!("Failed to write metadata: {}", e)))?;
349
350        Ok(())
351    }
352
353    /// Load index and metadata from disk
354    pub fn load<P: AsRef<Path>>(path: P, config: HnswConfig) -> Result<Self> {
355        let base_path = path.as_ref();
356
357        // Load HNSW index
358        let index_path = base_path.with_extension("index");
359        let hnsw = HnswIndex::load(&index_path, config)?;
360
361        // Load metadata
362        let meta_path = base_path.with_extension("meta");
363        let meta_json = std::fs::read_to_string(&meta_path)
364            .map_err(|e| CxpError::Search(format!("Failed to read metadata: {}", e)))?;
365        let metadata: HashMap<u64, EntryType> = serde_json::from_str(&meta_json)
366            .map_err(|e| CxpError::Search(format!("Failed to deserialize metadata: {}", e)))?;
367
368        Ok(Self { hnsw, metadata })
369    }

371    /// Set the search expansion parameter (ef_search)
372    ///
373    /// Higher values = better recall, slower search
374    pub fn set_expansion_search(&mut self, expansion: usize) {
375        self.hnsw.set_expansion_search(expansion);
376    }
377
378    /// Helper to attach metadata to search results
379    fn attach_metadata(
380        &self,
381        results: Vec<SearchResult>,
382    ) -> Result<Vec<SearchResultWithType>> {
383        results
384            .into_iter()
385            .map(|r| {
386                let entry_type = self
387                    .metadata
388                    .get(&r.id)
389                    .ok_or_else(|| {
390                        CxpError::Search(format!("Missing metadata for ID {}", r.id))
391                    })?
392                    .clone();
393
394                Ok(SearchResultWithType {
395                    id: r.id,
396                    distance: r.distance,
397                    entry_type,
398                })
399            })
400            .collect()
401    }
