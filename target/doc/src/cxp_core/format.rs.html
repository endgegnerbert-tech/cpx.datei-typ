1//! CXP File Format - ZIP container with structured content
2//!
3//! Structure:
4//! ```text
5//! file.cxp (ZIP)
6//! ├── manifest.msgpack     # Metadata & stats
7//! ├── file_map.msgpack     # File -> Chunk references
8//! ├── chunks/
9//! │   ├── 0001.zst         # Compressed chunks
10//! │   ├── 0002.zst
11//! │   └── ...
12//! ├── embeddings/          # Optional: Semantic search support
13//! │   ├── binary.bin       # Binary quantized embeddings
14//! │   ├── int8.bin         # Int8 quantized embeddings for rescoring
15//! │   └── index.hnsw       # HNSW index for fast search
16//! └── extensions/          # Optional app-specific data
17//!     └── ...
18//! ```

156    /// Scan the source directory for files
157    pub fn scan(&mut self) -> Result<&mut Self> {
158        tracing::info!("Scanning directory: {:?}", self.source_dir);
159
160        self.files = WalkDir::new(&self.source_dir)
161            .follow_links(true)
162            .into_iter()
163            .filter_map(|e| e.ok())
164            .filter(|e| e.file_type().is_file())
165            .filter(|e| {
166                // Filter by text extensions
167                e.path()
168                    .extension()
169                    .and_then(|ext| ext.to_str())
170                    .map(is_text_file)
171                    .unwrap_or(false)
172            })
173            .map(|e| e.path().to_path_buf())
174            .collect();
175
176        tracing::info!("Found {} text files to process", self.files.len());
177
178        // Scan for images if enabled
179        #[cfg(feature = "multimodal")]
180        if self.process_images {
181            self.image_files = WalkDir::new(&self.source_dir)
182                .follow_links(true)
183                .into_iter()
184                .filter_map(|e| e.ok())
185                .filter(|e| e.file_type().is_file())
186                .filter(|e| {
187                    // Filter by image extensions
188                    e.path()
189                        .extension()
190                        .and_then(|ext| ext.to_str())
191                        .map(is_image_file)
192                        .unwrap_or(false)
193                })
194                .map(|e| e.path().to_path_buf())
195                .collect();
196
197            tracing::info!("Found {} image files to process", self.image_files.len());
198        }
199
200        Ok(self)
201    }

203    /// Process all scanned files
204    pub fn process(&mut self) -> Result<&mut Self> {
205        let source_dir = self.source_dir.clone();
206
207        // Process text files and collect chunks
208        let results: Vec<_> = self.files
209            .iter()
210            .filter_map(|path| {
211                self.process_file(path, &source_dir).ok()
212            })
213            .collect();
214
215        // Add to chunk store and file map
216        for (entry, chunks) in results {
217            let chunk_refs = self.chunk_store.add_many(chunks);
218
219            // Update manifest with file type info
220            self.manifest.add_file_type(&entry.extension, &entry.path, entry.size);
221
222            // Store file entry with chunk refs
223            let entry_with_refs = FileEntry {
224                chunks: chunk_refs,
225                ..entry
226            };
227            self.file_map.files.insert(entry_with_refs.path.clone(), entry_with_refs);
228        }
229
230        // Process images if enabled (store as single chunks - whole image = 1 chunk)
231        #[cfg(feature = "multimodal")]
232        if self.process_images {
233            for path in &self.image_files.clone() {
234                if let Ok((entry, chunk)) = self.process_image(path, &source_dir) {
235                    // Create chunk ref before adding to store
236                    let chunk_ref = ChunkRef::from(&chunk);
237                    self.chunk_store.add(chunk);
238
239                    // Update manifest with file type info
240                    self.manifest.add_file_type(&entry.extension, &entry.path, entry.size);
241
242                    // Store file entry with chunk ref
243                    let entry_with_ref = FileEntry {
244                        chunks: vec![chunk_ref],
245                        is_image: true,
246                        ..entry
247                    };
248                    self.file_map.files.insert(entry_with_ref.path.clone(), entry_with_ref);
249                }
250            }
251
252            tracing::info!("Processed {} image files", self.image_files.len());
253        }
254
255        // Update manifest stats
256        let dedup_stats = self.chunk_store.stats();
257        self.manifest.stats.total_files = self.file_map.files.len();
258        self.manifest.stats.unique_chunks = dedup_stats.unique_chunks;
259        self.manifest.stats.original_size_bytes = dedup_stats.total_bytes as u64;
260        self.manifest.stats.dedup_savings_percent = dedup_stats.savings_percent();
261
262        tracing::info!(
263            "Processed {} files, {} unique chunks, {:.1}% dedup savings",
264            self.manifest.stats.total_files,
265            self.manifest.stats.unique_chunks,
266            self.manifest.stats.dedup_savings_percent
267        );
268
269        Ok(self)
270    }

345    /// Generate embeddings for all chunks
346    ///
347    /// This is automatically called during `build()` if embeddings are enabled.
348    /// You can also call it manually after `process()` to inspect the embeddings.
349    #[cfg(all(feature = "embeddings", feature = "search"))]
350    pub fn generate_embeddings(&mut self) -> Result<&mut Self> {
351        let engine = self.embedding_engine.as_ref()
352            .ok_or_else(|| CxpError::Embedding(
353                "Embedding engine not initialized. Call with_embeddings() first.".to_string()
354            ))?;
355
356        tracing::info!("Generating embeddings for {} unique chunks", self.chunk_store.len());
357
358        // Collect all chunk texts
359        let chunks: Vec<_> = self.chunk_store.chunks().collect();
360        let chunk_texts: Vec<&str> = chunks
361            .iter()
362            .map(|c| {
363                std::str::from_utf8(&c.data)
364                    .unwrap_or("[binary data]")
365            })
366            .collect();
367
368        // Process in batches to avoid OOM
369        const BATCH_SIZE: usize = 32;
370        let mut all_embeddings = Vec::new();
371
372        for batch in chunk_texts.chunks(BATCH_SIZE) {
373            let embeddings = engine.embed_batch(batch)?;
374            all_embeddings.extend(embeddings);
375        }
376
377        tracing::info!("Generated {} embeddings", all_embeddings.len());
378
379        // Create quantized embeddings
380        let quantized = QuantizedEmbeddings::from_floats(&all_embeddings);
381
382        tracing::info!(
383            "Quantized embeddings size: {:.2} MB (binary) + {:.2} MB (int8)",
384            quantized.binary.iter().map(|e| e.size_bytes()).sum::<usize>() as f64 / 1024.0 / 1024.0,
385            quantized.int8.iter().map(|e| e.size_bytes()).sum::<usize>() as f64 / 1024.0 / 1024.0
386        );
387
388        // Build HNSW index for binary embeddings
389        let config = HnswConfig::binary(engine.dimensions());
390        let mut index = HnswIndex::new(config)?;
391
392        tracing::info!("Building HNSW index...");
393
394        for (i, binary_emb) in quantized.binary.iter().enumerate() {
395            index.add_binary_embedding(i as u64, binary_emb)?;
396        }
397
398        tracing::info!("HNSW index built with {} vectors", index.len());
399
400        self.chunk_embeddings = Some(quantized);
401        self.search_index = Some(index);
402
403        Ok(self)
404    }

406    /// Generate multimodal embeddings for all chunks (text + images)
407    ///
408    /// This is automatically called during `build()` if multimodal embeddings are enabled.
409    /// Creates a UnifiedIndex with both text and image embeddings in the same vector space.
410    #[cfg(all(feature = "multimodal", feature = "search"))]
411    pub fn generate_multimodal_embeddings(&mut self) -> Result<&mut Self> {
412        tracing::info!("Generating multimodal embeddings for {} unique chunks", self.chunk_store.len());
413
414        // Process in batches to avoid OOM
415        const BATCH_SIZE: usize = 32;
416
417        // Step 1: Generate text embeddings
418        let chunks: Vec<_> = self.chunk_store.chunks().collect();
419        let chunk_texts: Vec<&str> = chunks
420            .iter()
421            .map(|c| {
422                std::str::from_utf8(&c.data)
423                    .unwrap_or("[binary data]")
424            })
425            .collect();
426
427        tracing::info!("Generating text embeddings for {} chunks...", chunk_texts.len());
428
429        let mut all_text_embeddings = Vec::new();
430        {
431            let engine = self.multimodal_engine.as_mut()
432                .ok_or_else(|| CxpError::Embedding(
433                    "Multimodal engine not initialized. Call with_multimodal_embeddings() first.".to_string()
434                ))?;
435
436            for batch in chunk_texts.chunks(BATCH_SIZE) {
437                let embeddings = engine.embed_batch_text(batch)?;
438                all_text_embeddings.extend(embeddings);
439            }
440        }
441
442        tracing::info!("Generated {} text embeddings", all_text_embeddings.len());
443
444        // Step 2: Generate image embeddings if needed
445        let mut all_image_embeddings = Vec::new();
446        let image_paths_strings: Vec<String> = if self.process_images && !self.image_files.is_empty() {
447            tracing::info!("Generating image embeddings for {} images...", self.image_files.len());
448
449            let image_paths: Vec<_> = self.image_files.iter().map(|p| p.as_path()).collect();
450
451            {
452                let engine = self.multimodal_engine.as_mut()
453                    .ok_or_else(|| CxpError::Embedding(
454                        "Multimodal engine not initialized.".to_string()
455                    ))?;
456
457                for batch in image_paths.chunks(BATCH_SIZE) {
458                    let embeddings = engine.embed_batch_images(batch)?;
459                    all_image_embeddings.extend(embeddings);
460                }
461            }
462
463            // Collect relative paths
464            self.image_files.iter()
465                .map(|img_path| {
466                    img_path
467                        .strip_prefix(&self.source_dir)
468                        .unwrap_or(img_path)
469                        .to_string_lossy()
470                        .to_string()
471                })
472                .collect()
473        } else {
474            Vec::new()
475        };
476
477        tracing::info!("Generated {} image embeddings", all_image_embeddings.len());
478
479        // Step 3: Build unified index with all embeddings
480        let config = HnswConfig::multimodal_float32();
481        let mut unified_index = UnifiedIndex::new(config)?;
482
483        let mut vector_id: u64 = 0;
484
485        // Add text embeddings to index
486        for (chunk_idx, embedding) in all_text_embeddings.into_iter().enumerate() {
487            let chunk_id = chunk_idx as u64;
488
489            // Find which file this chunk belongs to (for metadata)
490            let file_path = self.find_file_for_chunk_id(chunk_id)
491                .unwrap_or_else(|| "unknown".to_string());
492
493            unified_index.add_text(vector_id, &embedding, chunk_id, &file_path)?;
494            vector_id += 1;
495        }
496
497        // Add image embeddings to index
498        for (img_idx, embedding) in all_image_embeddings.into_iter().enumerate() {
499            if img_idx < image_paths_strings.len() {
500                let relative_path = &image_paths_strings[img_idx];
501                unified_index.add_image(vector_id, &embedding, relative_path)?;
502                vector_id += 1;
503            }
504        }
505
506        tracing::info!("Total vectors in unified index: {}", unified_index.len());
507        tracing::info!("  - Text chunks: {}", unified_index.text_count());
508        tracing::info!("  - Images: {}", unified_index.image_count());
509
510        self.unified_index = Some(unified_index);
511
512        Ok(self)
513    }

515    /// Find which file a chunk ID belongs to (helper for metadata)
516    #[cfg(all(feature = "multimodal", feature = "search"))]
517    fn find_file_for_chunk_id(&self, chunk_id: u64) -> Option<String> {
518        // Iterate through file map to find which file contains this chunk
519        for (path, entry) in &self.file_map.files {
520            if !entry.is_image {
521                // Check if this chunk ID is in the file's chunk range
522                // This is a simplified approach - in production you'd maintain a proper mapping
523                if chunk_id < entry.chunks.len() as u64 {
524                    return Some(path.clone());
525                }
526            }
527        }
528        None
529    }

531    /// Add an extension with data to this CXP file
532    ///
533    /// The extension will be registered and its data will be stored in the
534    /// extensions/{namespace}/ directory in the CXP file.
535    ///
536    /// # Arguments
537    /// * `ext` - An object implementing the Extension trait
538    /// * `data` - A HashMap of file names to their data (e.g., "conversations.msgpack" -> bytes)
539    ///
540    /// # Example
541    /// ```ignore
542    /// use std::collections::HashMap;
543    ///
544    /// let mut builder = CxpBuilder::new("./src");
545    /// let contextai = ContextAIExtension::new();
546    /// let data = contextai.to_extension_data()?;
547    ///
548    /// builder.add_extension(&contextai, data)?;
549    /// ```
550    pub fn add_extension<E: Extension + Clone>(
551        &mut self,
552        ext: &E,
553        data: HashMap<String, Vec<u8>>,
554    ) -> Result<&mut Self> {
555        // Register the extension
556        self.extension_manager.register(ext.clone());
557
558        // Add all data files
559        for (key, bytes) in data {
560            self.extension_manager.write_data(ext.namespace(), &key, &bytes)?;
561        }
562
563        // Update manifest to include this extension
564        if !self.manifest.extensions.contains(&ext.namespace().to_string()) {
565            self.manifest.extensions.push(ext.namespace().to_string());
566        }
567
568        tracing::info!(
569            "Added extension '{}' (v{}) with {} data files",
570            ext.namespace(),
571            ext.version(),
572            self.extension_manager.list_data_keys(ext.namespace()).len()
573        );
574
575        Ok(self)
576    }

578    /// Process a single file
579    fn process_file(&self, path: &Path, base_dir: &Path) -> Result<(FileEntry, Vec<Chunk>)> {
580        // Read file content
581        let mut file = File::open(path)?;
582        let metadata = file.metadata()?;
583        let mut content = Vec::with_capacity(metadata.len() as usize);
584        file.read_to_end(&mut content)?;
585
586        // Get relative path
587        let relative_path = path
588            .strip_prefix(base_dir)
589            .unwrap_or(path)
590            .to_string_lossy()
591            .to_string();
592
593        // Get extension
594        let extension = path
595            .extension()
596            .and_then(|e| e.to_str())
597            .unwrap_or("")
598            .to_lowercase();
599
600        // Chunk the content
601        let chunks = chunk_content(&content);
602
603        let entry = FileEntry {
604            path: relative_path,
605            extension,
606            size: metadata.len(),
607            chunks: Vec::new(), // Will be filled in with refs later
608            is_image: false,
609        };
610
611        Ok((entry, chunks))
612    }

615    #[cfg(feature = "multimodal")]
616    fn process_image(&self, path: &Path, base_dir: &Path) -> Result<(FileEntry, Chunk)> {
617        // Read image file
618        let mut file = File::open(path)?;
619        let metadata = file.metadata()?;
620        let mut content = Vec::with_capacity(metadata.len() as usize);
621        file.read_to_end(&mut content)?;
622
623        // Get relative path
624        let relative_path = path
625            .strip_prefix(base_dir)
626            .unwrap_or(path)
627            .to_string_lossy()
628            .to_string();
629
630        // Get extension
631        let extension = path
632            .extension()
633            .and_then(|e| e.to_str())
634            .unwrap_or("")
635            .to_lowercase();
636
637        // Create single chunk for entire image
638        let chunk = Chunk::new(content, 0);
639
640        let entry = FileEntry {
641            path: relative_path,
642            extension,
643            size: metadata.len(),
644            chunks: Vec::new(), // Will be filled in with ref later
645            is_image: true,
646        };
647
648        Ok((entry, chunk))
649    }

651    /// Build and write the CXP file
652    pub fn build<P: AsRef<Path>>(&mut self, output_path: P) -> Result<()> {
653        let output_path = output_path.as_ref();
654        tracing::info!("Building CXP file: {:?}", output_path);
655
656        // Generate embeddings if engine is set but embeddings haven't been generated yet
657        #[cfg(all(feature = "embeddings", feature = "search"))]
658        if self.embedding_engine.is_some() && self.chunk_embeddings.is_none() {
659            self.generate_embeddings()?;
660        }
661
662        // Generate multimodal embeddings if multimodal engine is set
663        #[cfg(all(feature = "multimodal", feature = "search"))]
664        if self.multimodal_engine.is_some() && self.unified_index.is_none() {
665            self.generate_multimodal_embeddings()?;
666        }
667
668        let file = File::create(output_path)?;
669        let mut zip = ZipWriter::new(file);
670
671        let options = FileOptions::<()>::default()
672            .compression_method(CompressionMethod::Stored); // We compress chunks ourselves
673
674        // Write manifest
675        let manifest_data = self.manifest.to_msgpack()?;
676        zip.start_file("manifest.msgpack", options.clone())?;
677        zip.write_all(&manifest_data)?;
678
679        // Write file map
680        let file_map_data = rmp_serde::to_vec(&self.file_map)?;
681        zip.start_file("file_map.msgpack", options.clone())?;
682        zip.write_all(&file_map_data)?;
683
684        // Write chunks
685        let chunks: Vec<_> = self.chunk_store.chunks().collect();
686        let total_chunks = chunks.len();
687
688        for (i, chunk) in chunks.iter().enumerate() {
689            let chunk_name = format!("chunks/{}.zst", chunk.id());
690            let compressed = compress(&chunk.data)?;
691
692            zip.start_file(&chunk_name, options.clone())?;
693            zip.write_all(&compressed)?;
694
695            if (i + 1) % 100 == 0 || i + 1 == total_chunks {
696                tracing::debug!("Written {}/{} chunks", i + 1, total_chunks);
697            }
698        }
699
700        // Write embeddings if present
701        #[cfg(all(feature = "embeddings", feature = "search"))]
702        if let Some(ref embeddings) = self.chunk_embeddings {
703            tracing::info!("Writing embeddings to CXP file...");
704
705            // Write binary embeddings
706            let binary_data = serialize_binary_embeddings(&embeddings.binary)?;
707            zip.start_file("embeddings/binary.bin", options.clone())?;
708            zip.write_all(&binary_data)?;
709
710            // Write int8 embeddings
711            let int8_data = serialize_int8_embeddings(&embeddings.int8)?;
712            zip.start_file("embeddings/int8.bin", options.clone())?;
713            zip.write_all(&int8_data)?;
714
715            // Mark that we have embeddings
716            if !self.manifest.extensions.contains(&"embeddings".to_string()) {
717                self.manifest.extensions.push("embeddings".to_string());
718            }
719
720            tracing::info!("Embeddings written successfully");
721        }
722
723        // Write HNSW index if present
724        #[cfg(all(feature = "embeddings", feature = "search"))]
725        if let Some(ref index) = self.search_index {
726            tracing::info!("Writing HNSW index to CXP file...");
727
728            // Save index to a temporary file first (USearch limitation)
729            let temp_dir = std::env::temp_dir();
730            let temp_index_path = temp_dir.join(format!("cxp_index_{}.hnsw", uuid::Uuid::new_v4()));
731
732            index.save(&temp_index_path)?;
733
734            // Read the index file and write to ZIP
735            let mut index_file = File::open(&temp_index_path)?;
736            let mut index_data = Vec::new();
737            index_file.read_to_end(&mut index_data)?;
738
739            zip.start_file("embeddings/index.hnsw", options.clone())?;
740            zip.write_all(&index_data)?;
741
742            // Clean up temp file
743            std::fs::remove_file(&temp_index_path)?;
744
745            tracing::info!("HNSW index written successfully ({} vectors)", index.len());
746        }
747
748        // Write UnifiedIndex if present (for multimodal)
749        #[cfg(all(feature = "multimodal", feature = "search"))]
750        if let Some(ref index) = self.unified_index {
751            tracing::info!("Writing UnifiedIndex to CXP file...");
752
753            // Save index to temporary files
754            let temp_dir = std::env::temp_dir();
755            let temp_base_path = temp_dir.join(format!("cxp_unified_{}", uuid::Uuid::new_v4()));
756
757            index.save(&temp_base_path)?;
758
759            // Read the index file and write to ZIP
760            let temp_index_path = temp_base_path.with_extension("index");
761            let mut index_file = File::open(&temp_index_path)?;
762            let mut index_data = Vec::new();
763            index_file.read_to_end(&mut index_data)?;
764
765            zip.start_file("embeddings/unified.index", options.clone())?;
766            zip.write_all(&index_data)?;
767
768            // Read the metadata file and write to ZIP
769            let temp_meta_path = temp_base_path.with_extension("meta");
770            let mut meta_file = File::open(&temp_meta_path)?;
771            let mut meta_data = Vec::new();
772            meta_file.read_to_end(&mut meta_data)?;
773
774            zip.start_file("embeddings/unified.meta", options.clone())?;
775            zip.write_all(&meta_data)?;
776
777            // Clean up temp files
778            std::fs::remove_file(&temp_index_path)?;
779            std::fs::remove_file(&temp_meta_path)?;
780
781            // Mark that we have embeddings
782            if !self.manifest.extensions.contains(&"embeddings".to_string()) {
783                self.manifest.extensions.push("embeddings".to_string());
784            }
785
786            tracing::info!("UnifiedIndex written successfully ({} vectors: {} text, {} images)",
787                index.len(), index.text_count(), index.image_count());
788        }
789
790        // Write extension data if present
791        if !self.extension_manager.list_extensions().is_empty() {
792            tracing::info!("Writing extension data to CXP file...");
793
794            // Write extension manifests
795            for manifest in self.extension_manager.manifests().values() {
796                let manifest_path = format!("extensions/{}/manifest.msgpack", manifest.namespace);
797                let manifest_data = manifest.to_msgpack()?;
798                zip.start_file(&manifest_path, options.clone())?;
799                zip.write_all(&manifest_data)?;
800            }
801
802            // Write extension data files
803            for (namespace, data_map) in self.extension_manager.all_data() {
804                for (key, data) in data_map {
805                    let data_path = format!("extensions/{}/{}", namespace, key);
806                    zip.start_file(&data_path, options.clone())?;
807                    zip.write_all(data)?;
808                }
809            }
810
811            tracing::info!(
812                "Written {} extensions to CXP file",
813                self.extension_manager.list_extensions().len()
814            );
815        }
816
817        zip.finish()?;
818
819        // Update manifest with final size
820        let final_size = std::fs::metadata(output_path)?.len();
821        self.manifest.stats.cxp_size_bytes = final_size;
822        self.manifest.stats.compression_ratio =
823            final_size as f64 / self.manifest.stats.original_size_bytes as f64;
824
825        tracing::info!(
826            "CXP file created: {:.2} MB (compression ratio: {:.2}%)",
827            final_size as f64 / 1024.0 / 1024.0,
828            self.manifest.stats.compression_ratio * 100.0
829        );
830
831        Ok(())
832    }
833}

836pub struct CxpReader {

838    pub manifest: Manifest,

842    archive_path: PathBuf,

844    extension_manager: ExtensionManager,

846    #[cfg(all(feature = "embeddings", feature = "search"))]
847    search_index: Option<HnswIndex>,

849    #[cfg(all(feature = "embeddings", feature = "search"))]
850    embeddings: Option<QuantizedEmbeddings>,

856impl CxpReader {
857    /// Open a CXP file for reading
858    pub fn open<P: AsRef<Path>>(path: P) -> Result<Self> {
859        let path = path.as_ref().to_path_buf();
860        let file = File::open(&path)?;
861        let mut archive = ZipArchive::new(file)?;
862
863        // Read manifest
864        let manifest = {
865            let mut manifest_file = archive.by_name("manifest.msgpack")?;
866            let mut data = Vec::new();
867            manifest_file.read_to_end(&mut data)?;
868            Manifest::from_msgpack(&data)?
869        };
870
871        // Read file map
872        let file_map = {
873            let mut file_map_file = archive.by_name("file_map.msgpack")?;
874            let mut data = Vec::new();
875            file_map_file.read_to_end(&mut data)?;
876            rmp_serde::from_slice(&data)?
877        };
878
879        // Load extension data if present
880        let mut extension_manager = ExtensionManager::new();
881
882        // Iterate through all files in the ZIP archive to find extensions
883        for i in 0..archive.len() {
884            let file = archive.by_index(i)?;
885            let file_name = file.name().to_string();
886
887            // Check if this is an extension file
888            if file_name.starts_with("extensions/") {
889                let parts: Vec<&str> = file_name.split('/').collect();
890                if parts.len() >= 3 {
891                    let namespace = parts[1];
892                    let file_key = parts[2..].join("/");
893
894                    // Read the file data
895                    let mut data = Vec::new();
896                    drop(file); // Close the borrowed file
897                    let mut file = archive.by_index(i)?;
898                    file.read_to_end(&mut data)?;
899
900                    if file_key == "manifest.msgpack" {
901                        // Load extension manifest
902                        if let Ok(ext_manifest) = crate::extensions::ExtensionManifest::from_msgpack(&data) {
903                            extension_manager.load_manifest(ext_manifest);
904                        }
905                    } else {
906                        // Load extension data
907                        extension_manager.load_data(namespace.to_string(), file_key, data);
908                    }
909                }
910            }
911        }
912
913        if !extension_manager.list_extensions().is_empty() {
914            tracing::info!(
915                "Loaded {} extensions from CXP file",
916                extension_manager.list_extensions().len()
917            );
918        }
919
920        Ok(Self {
921            manifest,
922            file_map,
923            archive_path: path,
924            extension_manager,
925            #[cfg(all(feature = "embeddings", feature = "search"))]
926            search_index: None,
927            #[cfg(all(feature = "embeddings", feature = "search"))]
928            embeddings: None,
929            #[cfg(all(feature = "multimodal", feature = "search"))]
930            unified_index: None,
931        })
932    }

944    /// Read a file's content by reconstructing from chunks
945    pub fn read_file(&self, path: &str) -> Result<Vec<u8>> {
946        let entry = self.file_map.files.get(path)
947            .ok_or_else(|| CxpError::FileNotFound(path.to_string()))?;
948
949        let file = File::open(&self.archive_path)?;
950        let mut archive = ZipArchive::new(file)?;
951
952        let mut content = Vec::with_capacity(entry.size as usize);
953
954        for chunk_ref in &entry.chunks {
955            let chunk_name = format!("chunks/{}.zst", &chunk_ref.hash[..16]);
956            let mut chunk_file = archive.by_name(&chunk_name)?;
957
958            let mut compressed = Vec::new();
959            chunk_file.read_to_end(&mut compressed)?;
960
961            let decompressed = decompress(&compressed)?;
962            content.extend_from_slice(&decompressed);
963        }
964
965        Ok(content)
966    }

1183    /// Perform semantic search using a query embedding
1184    ///
1185    /// Returns the top-k most similar chunks by ID.
1186    /// You must call `load_embeddings()` first.
1187    ///
1188    /// # Arguments
1189    /// * `query_embedding` - The query vector (should match the model's dimensions)
1190    /// * `top_k` - Number of results to return
1191    ///
1192    /// # Returns
1193    /// Vector of (chunk_id, similarity_score) tuples, sorted by similarity (highest first)
1194    #[cfg(all(feature = "embeddings", feature = "search"))]
1195    pub fn search_semantic(
1196        &self,
1197        query_embedding: &[f32],
1198        top_k: usize,
1199    ) -> Result<Vec<SearchResult>> {
1200        let index = self.search_index.as_ref()
1201            .ok_or_else(|| CxpError::Search(
1202                "Embeddings not loaded. Call load_embeddings() first.".to_string()
1203            ))?;
1204
1205        let embeddings = self.embeddings.as_ref()
1206            .ok_or_else(|| CxpError::Search(
1207                "Embeddings not loaded. Call load_embeddings() first.".to_string()
1208            ))?;
1209
1210        // Convert query to binary for fast initial search
1211        let query_binary = BinaryEmbedding::from_float(query_embedding);
1212
1213        // Search with HNSW (binary)
1214        let candidates = index.search_binary_embedding(&query_binary, top_k * 2)?;
1215
1216        // Rescore with Int8 for better accuracy
1217        let query_int8 = Int8Embedding::from_float(query_embedding);
1218
1219        let mut rescored: Vec<_> = candidates
1220            .iter()
1221            .map(|result| {
1222                let chunk_id = result.id as usize;
1223                let score = if chunk_id < embeddings.int8.len() {
1224                    embeddings.int8[chunk_id].dot_product(&query_int8)
1225                } else {
1226                    0.0
1227                };
1228                SearchResult {
1229                    id: result.id,
1230                    distance: -score,  // Negate for sorting (higher is better)
1231                }
1232            })
1233            .collect();
1234
1235        // Sort by score (descending)
1236        rescored.sort_by(|a, b| a.distance.partial_cmp(&b.distance).unwrap());
1237
1238        // Take top-k and fix the distance sign
1239        Ok(rescored
1240            .into_iter()
1241            .take(top_k)
1242            .map(|mut r| {
1243                r.distance = -r.distance;
1244                r
1245            })
1246            .collect())
1247    }

1249    /// Perform multimodal semantic search with type filtering
1250    ///
1251    /// Searches across both text and images using the UnifiedIndex.
1252    /// You must call `load_unified_index()` first.
1253    ///
1254    /// # Arguments
1255    /// * `query_embedding` - The query vector (should match the model's dimensions)
1256    /// * `top_k` - Number of results to return
1257    /// * `result_type` - Filter by type: "text", "image", or "all"
1258    ///
1259    /// # Returns
1260    /// Vector of search results with type information, sorted by similarity (highest first)
1261    #[cfg(all(feature = "multimodal", feature = "search"))]
1262    pub fn search_multimodal(
1263        &self,
1264        query_embedding: &[f32],
1265        top_k: usize,
1266        result_type: &str,
1267    ) -> Result<Vec<crate::SearchResultWithType>> {
1268        let index = self.unified_index.as_ref()
1269            .ok_or_else(|| CxpError::Search(
1270                "UnifiedIndex not loaded. Call load_unified_index() first.".to_string()
1271            ))?;
1272
1273        // Search based on result type filter
1274        let results = match result_type.to_lowercase().as_str() {
1275            "text" => index.search_text_only(query_embedding, top_k)?,
1276            "image" => index.search_images_only(query_embedding, top_k)?,
1277            "all" | _ => index.search(query_embedding, top_k)?,
1278        };

1281    }

1319    /// Get chunk text by ID
1320    ///
1321    /// This is useful for retrieving the actual content of chunks found by semantic search.
1322    #[cfg(all(feature = "embeddings", feature = "search"))]
1323    pub fn get_chunk_text(&self, chunk_id: u64) -> Result<String> {
1324        let file = File::open(&self.archive_path)?;
1325        let mut archive = ZipArchive::new(file)?;
1326
1327        // We need to find the chunk by iterating through the chunk store
1328        // For now, we'll use the chunk hash format
1329        let chunk_name = format!("chunks/{:016x}.zst", chunk_id);
1330
1331        let mut chunk_file = archive.by_name(&chunk_name)
1332            .or_else(|_| {
1333                // Try alternative naming if the first format doesn't work
1334                // This is a fallback - in practice you'd maintain a chunk ID -> hash mapping
1335                Err(CxpError::FileNotFound(format!("Chunk {} not found", chunk_id)))
1336            })?;
1337
1338        let mut compressed = Vec::new();
1339        chunk_file.read_to_end(&mut compressed)?;
1340
1341        let decompressed = decompress(&compressed)?;
1342
1343        String::from_utf8(decompressed)
1344            .map_err(|e| CxpError::Serialization(format!("Invalid UTF-8 in chunk: {}", e)))
1345    }
