1//! CXP File Format - ZIP container with structured content
2//!
3//! Structure:
4//! ```text
5//! file.cxp (ZIP)
6//! ├── manifest.msgpack     # Metadata & stats
7//! ├── file_map.msgpack     # File -> Chunk references
8//! ├── chunks/
9//! │   ├── 0001.zst         # Compressed chunks
10//! │   ├── 0002.zst
11//! │   └── ...
12//! ├── embeddings/          # Optional: Semantic search support
13//! │   ├── binary.bin       # Binary quantized embeddings
14//! │   ├── int8.bin         # Int8 quantized embeddings for rescoring
15//! │   └── index.hnsw       # HNSW index for fast search
16//! └── extensions/          # Optional app-specific data
17//!     └── ...
18//! ```

122    /// Scan the source directory for files
123    pub fn scan(&mut self) -> Result<&mut Self> {
124        tracing::info!("Scanning directory: {:?}", self.source_dir);
125
126        self.files = WalkDir::new(&self.source_dir)
127            .follow_links(true)
128            .into_iter()
129            .filter_map(|e| e.ok())
130            .filter(|e| e.file_type().is_file())
131            .filter(|e| {
132                // Filter by text extensions
133                e.path()
134                    .extension()
135                    .and_then(|ext| ext.to_str())
136                    .map(is_text_file)
137                    .unwrap_or(false)
138            })
139            .map(|e| e.path().to_path_buf())
140            .collect();
141
142        tracing::info!("Found {} files to process", self.files.len());
143        Ok(self)
144    }

146    /// Process all scanned files
147    pub fn process(&mut self) -> Result<&mut Self> {
148        let source_dir = self.source_dir.clone();
149
150        // Process files and collect chunks
151        let results: Vec<_> = self.files
152            .iter()
153            .filter_map(|path| {
154                self.process_file(path, &source_dir).ok()
155            })
156            .collect();
157
158        // Add to chunk store and file map
159        for (entry, chunks) in results {
160            let chunk_refs = self.chunk_store.add_many(chunks);
161
162            // Update manifest with file type info
163            self.manifest.add_file_type(&entry.extension, &entry.path, entry.size);
164
165            // Store file entry with chunk refs
166            let entry_with_refs = FileEntry {
167                chunks: chunk_refs,
168                ..entry
169            };
170            self.file_map.files.insert(entry_with_refs.path.clone(), entry_with_refs);
171        }
172
173        // Update manifest stats
174        let dedup_stats = self.chunk_store.stats();
175        self.manifest.stats.total_files = self.file_map.files.len();
176        self.manifest.stats.unique_chunks = dedup_stats.unique_chunks;
177        self.manifest.stats.original_size_bytes = dedup_stats.total_bytes as u64;
178        self.manifest.stats.dedup_savings_percent = dedup_stats.savings_percent();
179
180        tracing::info!(
181            "Processed {} files, {} unique chunks, {:.1}% dedup savings",
182            self.manifest.stats.total_files,
183            self.manifest.stats.unique_chunks,
184            self.manifest.stats.dedup_savings_percent
185        );
186
187        Ok(self)
188    }

190    /// Enable embedding generation (requires both "embeddings" and "search" features)
191    ///
192    /// This loads an embedding model and will generate embeddings for all chunks
193    /// during the build process. The embeddings are stored in the CXP file along
194    /// with a HNSW index for fast semantic search.
195    ///
196    /// # Arguments
197    /// * `model_path` - Path to the directory containing model.onnx and tokenizer.json
198    /// * `model` - The embedding model to use (e.g., EmbeddingModel::MiniLM)
199    ///
200    /// # Example
201    /// ```ignore
202    /// builder
203    ///     .scan()?
204    ///     .with_embeddings("./models/all-MiniLM-L6-v2", EmbeddingModel::MiniLM)?
205    ///     .process()?
206    ///     .build("output.cxp")?;
207    /// ```
208    #[cfg(all(feature = "embeddings", feature = "search"))]
209    pub fn with_embeddings<P: AsRef<Path>>(
210        &mut self,
211        model_path: P,
212        model: EmbeddingModel,
213    ) -> Result<&mut Self> {
214        tracing::info!("Loading embedding model: {}", model.name());
215
216        let engine = EmbeddingEngine::load(model_path, model)?;
217
218        // Update manifest with embedding info
219        self.manifest.embedding_model = Some(model.name().to_string());
220        self.manifest.embedding_dim = Some(model.dimensions());
221
222        self.embedding_engine = Some(engine);
223
224        Ok(self)
225    }

227    /// Generate embeddings for all chunks
228    ///
229    /// This is automatically called during `build()` if embeddings are enabled.
230    /// You can also call it manually after `process()` to inspect the embeddings.
231    #[cfg(all(feature = "embeddings", feature = "search"))]
232    pub fn generate_embeddings(&mut self) -> Result<&mut Self> {
233        let engine = self.embedding_engine.as_ref()
234            .ok_or_else(|| CxpError::Embedding(
235                "Embedding engine not initialized. Call with_embeddings() first.".to_string()
236            ))?;
237
238        tracing::info!("Generating embeddings for {} unique chunks", self.chunk_store.len());
239
240        // Collect all chunk texts
241        let chunks: Vec<_> = self.chunk_store.chunks().collect();
242        let chunk_texts: Vec<&str> = chunks
243            .iter()
244            .map(|c| {
245                std::str::from_utf8(&c.data)
246                    .unwrap_or("[binary data]")
247            })
248            .collect();
249
250        // Process in batches to avoid OOM
251        const BATCH_SIZE: usize = 32;
252        let mut all_embeddings = Vec::new();
253
254        for batch in chunk_texts.chunks(BATCH_SIZE) {
255            let embeddings = engine.embed_batch(batch)?;
256            all_embeddings.extend(embeddings);
257        }
258
259        tracing::info!("Generated {} embeddings", all_embeddings.len());
260
261        // Create quantized embeddings
262        let quantized = QuantizedEmbeddings::from_floats(&all_embeddings);
263
264        tracing::info!(
265            "Quantized embeddings size: {:.2} MB (binary) + {:.2} MB (int8)",
266            quantized.binary.iter().map(|e| e.size_bytes()).sum::<usize>() as f64 / 1024.0 / 1024.0,
267            quantized.int8.iter().map(|e| e.size_bytes()).sum::<usize>() as f64 / 1024.0 / 1024.0
268        );
269
270        // Build HNSW index for binary embeddings
271        let config = HnswConfig::binary(engine.dimensions());
272        let mut index = HnswIndex::new(config)?;
273
274        tracing::info!("Building HNSW index...");
275
276        for (i, binary_emb) in quantized.binary.iter().enumerate() {
277            index.add_binary_embedding(i as u64, binary_emb)?;
278        }
279
280        tracing::info!("HNSW index built with {} vectors", index.len());
281
282        self.chunk_embeddings = Some(quantized);
283        self.search_index = Some(index);
284
285        Ok(self)
286    }

288    /// Add an extension with data to this CXP file
289    ///
290    /// The extension will be registered and its data will be stored in the
291    /// extensions/{namespace}/ directory in the CXP file.
292    ///
293    /// # Arguments
294    /// * `ext` - An object implementing the Extension trait
295    /// * `data` - A HashMap of file names to their data (e.g., "conversations.msgpack" -> bytes)
296    ///
297    /// # Example
298    /// ```ignore
299    /// use std::collections::HashMap;
300    ///
301    /// let mut builder = CxpBuilder::new("./src");
302    /// let contextai = ContextAIExtension::new();
303    /// let data = contextai.to_extension_data()?;
304    ///
305    /// builder.add_extension(&contextai, data)?;
306    /// ```
307    pub fn add_extension<E: Extension + Clone>(
308        &mut self,
309        ext: &E,
310        data: HashMap<String, Vec<u8>>,
311    ) -> Result<&mut Self> {
312        // Register the extension
313        self.extension_manager.register(ext.clone());
314
315        // Add all data files
316        for (key, bytes) in data {
317            self.extension_manager.write_data(ext.namespace(), &key, &bytes)?;
318        }
319
320        // Update manifest to include this extension
321        if !self.manifest.extensions.contains(&ext.namespace().to_string()) {
322            self.manifest.extensions.push(ext.namespace().to_string());
323        }
324
325        tracing::info!(
326            "Added extension '{}' (v{}) with {} data files",
327            ext.namespace(),
328            ext.version(),
329            self.extension_manager.list_data_keys(ext.namespace()).len()
330        );
331
332        Ok(self)
333    }

335    /// Process a single file
336    fn process_file(&self, path: &Path, base_dir: &Path) -> Result<(FileEntry, Vec<Chunk>)> {
337        // Read file content
338        let mut file = File::open(path)?;
339        let metadata = file.metadata()?;
340        let mut content = Vec::with_capacity(metadata.len() as usize);
341        file.read_to_end(&mut content)?;
342
343        // Get relative path
344        let relative_path = path
345            .strip_prefix(base_dir)
346            .unwrap_or(path)
347            .to_string_lossy()
348            .to_string();
349
350        // Get extension
351        let extension = path
352            .extension()
353            .and_then(|e| e.to_str())
354            .unwrap_or("")
355            .to_lowercase();
356
357        // Chunk the content
358        let chunks = chunk_content(&content);
359
360        let entry = FileEntry {
361            path: relative_path,
362            extension,
363            size: metadata.len(),
364            chunks: Vec::new(), // Will be filled in with refs later
365        };
366
367        Ok((entry, chunks))
368    }

370    /// Build and write the CXP file
371    pub fn build<P: AsRef<Path>>(&mut self, output_path: P) -> Result<()> {
372        let output_path = output_path.as_ref();
373        tracing::info!("Building CXP file: {:?}", output_path);
374
375        // Generate embeddings if engine is set but embeddings haven't been generated yet
376        #[cfg(all(feature = "embeddings", feature = "search"))]
377        if self.embedding_engine.is_some() && self.chunk_embeddings.is_none() {
378            self.generate_embeddings()?;
379        }
380
381        let file = File::create(output_path)?;
382        let mut zip = ZipWriter::new(file);
383
384        let options = FileOptions::<()>::default()
385            .compression_method(CompressionMethod::Stored); // We compress chunks ourselves
386
387        // Write manifest
388        let manifest_data = self.manifest.to_msgpack()?;
389        zip.start_file("manifest.msgpack", options.clone())?;
390        zip.write_all(&manifest_data)?;
391
392        // Write file map
393        let file_map_data = rmp_serde::to_vec(&self.file_map)?;
394        zip.start_file("file_map.msgpack", options.clone())?;
395        zip.write_all(&file_map_data)?;
396
397        // Write chunks
398        let chunks: Vec<_> = self.chunk_store.chunks().collect();
399        let total_chunks = chunks.len();
400
401        for (i, chunk) in chunks.iter().enumerate() {
402            let chunk_name = format!("chunks/{}.zst", chunk.id());
403            let compressed = compress(&chunk.data)?;
404
405            zip.start_file(&chunk_name, options.clone())?;
406            zip.write_all(&compressed)?;
407
408            if (i + 1) % 100 == 0 || i + 1 == total_chunks {
409                tracing::debug!("Written {}/{} chunks", i + 1, total_chunks);
410            }
411        }
412
413        // Write embeddings if present
414        #[cfg(all(feature = "embeddings", feature = "search"))]
415        if let Some(ref embeddings) = self.chunk_embeddings {
416            tracing::info!("Writing embeddings to CXP file...");
417
418            // Write binary embeddings
419            let binary_data = serialize_binary_embeddings(&embeddings.binary)?;
420            zip.start_file("embeddings/binary.bin", options.clone())?;
421            zip.write_all(&binary_data)?;
422
423            // Write int8 embeddings
424            let int8_data = serialize_int8_embeddings(&embeddings.int8)?;
425            zip.start_file("embeddings/int8.bin", options.clone())?;
426            zip.write_all(&int8_data)?;
427
428            // Mark that we have embeddings
429            if !self.manifest.extensions.contains(&"embeddings".to_string()) {
430                self.manifest.extensions.push("embeddings".to_string());
431            }
432
433            tracing::info!("Embeddings written successfully");
434        }
435
436        // Write HNSW index if present
437        #[cfg(all(feature = "embeddings", feature = "search"))]
438        if let Some(ref index) = self.search_index {
439            tracing::info!("Writing HNSW index to CXP file...");
440
441            // Save index to a temporary file first (USearch limitation)
442            let temp_dir = std::env::temp_dir();
443            let temp_index_path = temp_dir.join(format!("cxp_index_{}.hnsw", uuid::Uuid::new_v4()));
444
445            index.save(&temp_index_path)?;
446
447            // Read the index file and write to ZIP
448            let mut index_file = File::open(&temp_index_path)?;
449            let mut index_data = Vec::new();
450            index_file.read_to_end(&mut index_data)?;
451
452            zip.start_file("embeddings/index.hnsw", options.clone())?;
453            zip.write_all(&index_data)?;
454
455            // Clean up temp file
456            std::fs::remove_file(&temp_index_path)?;
457
458            tracing::info!("HNSW index written successfully ({} vectors)", index.len());
459        }
460
461        // Write extension data if present
462        if !self.extension_manager.list_extensions().is_empty() {
463            tracing::info!("Writing extension data to CXP file...");
464
465            // Write extension manifests
466            for manifest in self.extension_manager.manifests().values() {
467                let manifest_path = format!("extensions/{}/manifest.msgpack", manifest.namespace);
468                let manifest_data = manifest.to_msgpack()?;
469                zip.start_file(&manifest_path, options.clone())?;
470                zip.write_all(&manifest_data)?;
471            }
472
473            // Write extension data files
474            for (namespace, data_map) in self.extension_manager.all_data() {
475                for (key, data) in data_map {
476                    let data_path = format!("extensions/{}/{}", namespace, key);
477                    zip.start_file(&data_path, options.clone())?;
478                    zip.write_all(data)?;
479                }
480            }
481
482            tracing::info!(
483                "Written {} extensions to CXP file",
484                self.extension_manager.list_extensions().len()
485            );
486        }
487
488        zip.finish()?;
489
490        // Update manifest with final size
491        let final_size = std::fs::metadata(output_path)?.len();
492        self.manifest.stats.cxp_size_bytes = final_size;
493        self.manifest.stats.compression_ratio =
494            final_size as f64 / self.manifest.stats.original_size_bytes as f64;
495
496        tracing::info!(
497            "CXP file created: {:.2} MB (compression ratio: {:.2}%)",
498            final_size as f64 / 1024.0 / 1024.0,
499            self.manifest.stats.compression_ratio * 100.0
500        );
501
502        Ok(())
503    }

506/// Reader for CXP files
507pub struct CxpReader {
508    /// The manifest
509    pub manifest: Manifest,
510    /// File map
511    pub file_map: FileMap,
512    /// ZIP archive handle
513    archive_path: PathBuf,
514    /// Extension manager for reading app-specific data
515    extension_manager: ExtensionManager,
516    /// Cached HNSW index for semantic search
517    #[cfg(all(feature = "embeddings", feature = "search"))]
518    search_index: Option<HnswIndex>,
519    /// Cached embeddings for rescoring
520    #[cfg(all(feature = "embeddings", feature = "search"))]
521    embeddings: Option<QuantizedEmbeddings>,
522}
523
524impl CxpReader {
525    /// Open a CXP file for reading
526    pub fn open<P: AsRef<Path>>(path: P) -> Result<Self> {
527        let path = path.as_ref().to_path_buf();
528        let file = File::open(&path)?;
529        let mut archive = ZipArchive::new(file)?;
530
531        // Read manifest
532        let manifest = {
533            let mut manifest_file = archive.by_name("manifest.msgpack")?;
534            let mut data = Vec::new();
535            manifest_file.read_to_end(&mut data)?;
536            Manifest::from_msgpack(&data)?
537        };
538
539        // Read file map
540        let file_map = {
541            let mut file_map_file = archive.by_name("file_map.msgpack")?;
542            let mut data = Vec::new();
543            file_map_file.read_to_end(&mut data)?;
544            rmp_serde::from_slice(&data)?
545        };
546
547        // Load extension data if present
548        let mut extension_manager = ExtensionManager::new();
549
550        // Iterate through all files in the ZIP archive to find extensions
551        for i in 0..archive.len() {
552            let file = archive.by_index(i)?;
553            let file_name = file.name().to_string();
554
555            // Check if this is an extension file
556            if file_name.starts_with("extensions/") {
557                let parts: Vec<&str> = file_name.split('/').collect();
558                if parts.len() >= 3 {
559                    let namespace = parts[1];
560                    let file_key = parts[2..].join("/");
561
562                    // Read the file data
563                    let mut data = Vec::new();
564                    drop(file); // Close the borrowed file
565                    let mut file = archive.by_index(i)?;
566                    file.read_to_end(&mut data)?;
567
568                    if file_key == "manifest.msgpack" {
569                        // Load extension manifest
570                        if let Ok(ext_manifest) = crate::extensions::ExtensionManifest::from_msgpack(&data) {
571                            extension_manager.load_manifest(ext_manifest);
572                        }
573                    } else {
574                        // Load extension data
575                        extension_manager.load_data(namespace.to_string(), file_key, data);
576                    }
577                }
578            }
579        }
580
581        if !extension_manager.list_extensions().is_empty() {
582            tracing::info!(
583                "Loaded {} extensions from CXP file",
584                extension_manager.list_extensions().len()
585            );
586        }
587
588        Ok(Self {
589            manifest,
590            file_map,
591            archive_path: path,
592            extension_manager,
593            #[cfg(all(feature = "embeddings", feature = "search"))]
594            search_index: None,
595            #[cfg(all(feature = "embeddings", feature = "search"))]
596            embeddings: None,
597        })
598    }

610    /// Read a file's content by reconstructing from chunks
611    pub fn read_file(&self, path: &str) -> Result<Vec<u8>> {
612        let entry = self.file_map.files.get(path)
613            .ok_or_else(|| CxpError::FileNotFound(path.to_string()))?;
614
615        let file = File::open(&self.archive_path)?;
616        let mut archive = ZipArchive::new(file)?;
617
618        let mut content = Vec::with_capacity(entry.size as usize);
619
620        for chunk_ref in &entry.chunks {
621            let chunk_name = format!("chunks/{}.zst", &chunk_ref.hash[..16]);
622            let mut chunk_file = archive.by_name(&chunk_name)?;
623
624            let mut compressed = Vec::new();
625            chunk_file.read_to_end(&mut compressed)?;
626
627            let decompressed = decompress(&compressed)?;
628            content.extend_from_slice(&decompressed);
629        }
630
631        Ok(content)
632    }

647    /// Load embeddings as an EmbeddingStore without caching
648    ///
649    /// Returns the embeddings without loading the HNSW index or caching them.
650    /// Use this if you only need to access the embeddings directly.
651    #[cfg(any(feature = "embeddings", feature = "embeddings-wasm"))]
652    pub fn get_embedding_store(&self) -> Result<crate::EmbeddingStore> {
653        if !self.has_embeddings() {
654            return Err(CxpError::Embedding(
655                "This CXP file does not contain embeddings".to_string()
656            ));
657        }
658
659        tracing::info!("Loading embeddings from CXP file...");
660
661        let file = File::open(&self.archive_path)?;
662        let mut archive = ZipArchive::new(file)?;
663
664        // Load binary embeddings
665        let binary_embeddings = {
666            let mut file = archive.by_name("embeddings/binary.bin")?;
667            let mut data = Vec::new();
668            file.read_to_end(&mut data)?;
669            deserialize_binary_embeddings(&data)?
670        };
671
672        // Load int8 embeddings
673        let int8_embeddings = {
674            let mut file = archive.by_name("embeddings/int8.bin")?;
675            let mut data = Vec::new();
676            file.read_to_end(&mut data)?;
677            deserialize_int8_embeddings(&data)?
678        };
679
680        let dimensions = self.manifest.embedding_dim
681            .ok_or_else(|| CxpError::Embedding("No embedding dimension in manifest".to_string()))?;
682
683        tracing::info!("Loaded {} embeddings", binary_embeddings.len());
684
685        Ok(crate::EmbeddingStore::new(
686            binary_embeddings,
687            int8_embeddings,
688            dimensions,
689        ))
690    }

692    /// Load embeddings and search index into memory
693    ///
694    /// This must be called before using semantic search functions.
695    /// The embeddings and index are cached for subsequent searches.
696    #[cfg(all(feature = "embeddings", feature = "search"))]
697    pub fn load_embeddings(&mut self) -> Result<()> {
698        if !self.has_embeddings() {
699            return Err(CxpError::Embedding(
700                "This CXP file does not contain embeddings".to_string()
701            ));
702        }
703
704        if self.search_index.is_some() {
705            return Ok(());  // Already loaded
706        }
707
708        tracing::info!("Loading embeddings from CXP file...");
709
710        let file = File::open(&self.archive_path)?;
711        let mut archive = ZipArchive::new(file)?;
712
713        // Load binary embeddings
714        let binary_embeddings = {
715            let mut file = archive.by_name("embeddings/binary.bin")?;
716            let mut data = Vec::new();
717            file.read_to_end(&mut data)?;
718            deserialize_binary_embeddings(&data)?
719        };
720
721        // Load int8 embeddings
722        let int8_embeddings = {
723            let mut file = archive.by_name("embeddings/int8.bin")?;
724            let mut data = Vec::new();
725            file.read_to_end(&mut data)?;
726            deserialize_int8_embeddings(&data)?
727        };
728
729        tracing::info!("Loaded {} embeddings", binary_embeddings.len());
730
731        self.embeddings = Some(QuantizedEmbeddings {
732            binary: binary_embeddings,
733            int8: int8_embeddings,
734        });
735
736        // Load HNSW index
737        let file = File::open(&self.archive_path)?;
738        let mut archive = ZipArchive::new(file)?;
739
740        let mut index_file = archive.by_name("embeddings/index.hnsw")?;
741        let mut index_data = Vec::new();
742        index_file.read_to_end(&mut index_data)?;
743
744        // Save to temp file (USearch limitation)
745        let temp_dir = std::env::temp_dir();
746        let temp_index_path = temp_dir.join(format!("cxp_index_{}.hnsw", uuid::Uuid::new_v4()));
747
748        let mut temp_file = File::create(&temp_index_path)?;
749        temp_file.write_all(&index_data)?;
750        drop(temp_file);
751
752        // Load index
753        let dimensions = self.manifest.embedding_dim
754            .ok_or_else(|| CxpError::Embedding("No embedding dimension in manifest".to_string()))?;
755
756        let config = HnswConfig::binary(dimensions);
757        let index = HnswIndex::load(&temp_index_path, config)?;
758
759        // Clean up temp file
760        std::fs::remove_file(&temp_index_path)?;
761
762        tracing::info!("Loaded HNSW index with {} vectors", index.len());
763
764        self.search_index = Some(index);
765
766        Ok(())
767    }

769    /// Perform semantic search using a query embedding
770    ///
771    /// Returns the top-k most similar chunks by ID.
772    /// You must call `load_embeddings()` first.
773    ///
774    /// # Arguments
775    /// * `query_embedding` - The query vector (should match the model's dimensions)
776    /// * `top_k` - Number of results to return
777    ///
778    /// # Returns
779    /// Vector of (chunk_id, similarity_score) tuples, sorted by similarity (highest first)
780    #[cfg(all(feature = "embeddings", feature = "search"))]
781    pub fn search_semantic(
782        &self,
783        query_embedding: &[f32],
784        top_k: usize,
785    ) -> Result<Vec<SearchResult>> {
786        let index = self.search_index.as_ref()
787            .ok_or_else(|| CxpError::Search(
788                "Embeddings not loaded. Call load_embeddings() first.".to_string()
789            ))?;
790
791        let embeddings = self.embeddings.as_ref()
792            .ok_or_else(|| CxpError::Search(
793                "Embeddings not loaded. Call load_embeddings() first.".to_string()
794            ))?;
795
796        // Convert query to binary for fast initial search
797        let query_binary = BinaryEmbedding::from_float(query_embedding);
798
799        // Search with HNSW (binary)
800        let candidates = index.search_binary_embedding(&query_binary, top_k * 2)?;
801
802        // Rescore with Int8 for better accuracy
803        let query_int8 = Int8Embedding::from_float(query_embedding);
804
805        let mut rescored: Vec<_> = candidates
806            .iter()
807            .map(|result| {
808                let chunk_id = result.id as usize;
809                let score = if chunk_id < embeddings.int8.len() {
810                    embeddings.int8[chunk_id].dot_product(&query_int8)
811                } else {
812                    0.0
813                };
814                SearchResult {
815                    id: result.id,
816                    distance: -score,  // Negate for sorting (higher is better)
817                }
818            })
819            .collect();
820
821        // Sort by score (descending)
822        rescored.sort_by(|a, b| a.distance.partial_cmp(&b.distance).unwrap());
823
824        // Take top-k and fix the distance sign
825        Ok(rescored
826            .into_iter()
827            .take(top_k)
828            .map(|mut r| {
829                r.distance = -r.distance;
830                r
831            })
832            .collect())
833    }

835    /// Get chunk text by ID
836    ///
837    /// This is useful for retrieving the actual content of chunks found by semantic search.
838    #[cfg(all(feature = "embeddings", feature = "search"))]
839    pub fn get_chunk_text(&self, chunk_id: u64) -> Result<String> {
840        let file = File::open(&self.archive_path)?;
841        let mut archive = ZipArchive::new(file)?;
842
843        // We need to find the chunk by iterating through the chunk store
844        // For now, we'll use the chunk hash format
845        let chunk_name = format!("chunks/{:016x}.zst", chunk_id);
846
847        let mut chunk_file = archive.by_name(&chunk_name)
848            .or_else(|_| {
849                // Try alternative naming if the first format doesn't work
850                // This is a fallback - in practice you'd maintain a chunk ID -> hash mapping
851                Err(CxpError::FileNotFound(format!("Chunk {} not found", chunk_id)))
852            })?;
853
854        let mut compressed = Vec::new();
855        chunk_file.read_to_end(&mut compressed)?;
856
857        let decompressed = decompress(&compressed)?;
858
859        String::from_utf8(decompressed)
860            .map_err(|e| CxpError::Serialization(format!("Invalid UTF-8 in chunk: {}", e)))
861    }
