1//! CXP File Format - ZIP container with structured content
2//!
3//! Structure:
4//! ```text
5//! file.cxp (ZIP)
6//! ├── manifest.msgpack     # Metadata & stats
7//! ├── file_map.msgpack     # File -> Chunk references
8//! ├── chunks/
9//! │   ├── 0001.zst         # Compressed chunks
10//! │   ├── 0002.zst
11//! │   └── ...
12//! ├── embeddings/          # Optional: Semantic search support
13//! │   ├── binary.bin       # Binary quantized embeddings
14//! │   ├── int8.bin         # Int8 quantized embeddings for rescoring
15//! │   └── index.hnsw       # HNSW index for fast search
16//! └── extensions/          # Optional app-specific data
17//!     └── ...
18//! ```

148    /// Scan the source directory for files
149    pub fn scan(&mut self) -> Result<&mut Self> {
150        tracing::info!("Scanning directory: {:?}", self.source_dir);
151
152        self.files = WalkDir::new(&self.source_dir)
153            .follow_links(true)
154            .into_iter()
155            .filter_map(|e| e.ok())
156            .filter(|e| e.file_type().is_file())
157            .filter(|e| {
158                // Filter by text extensions
159                e.path()
160                    .extension()
161                    .and_then(|ext| ext.to_str())
162                    .map(is_text_file)
163                    .unwrap_or(false)
164            })
165            .map(|e| e.path().to_path_buf())
166            .collect();
167
168        tracing::info!("Found {} text files to process", self.files.len());
169
170        // Scan for images if enabled
171        #[cfg(feature = "multimodal")]
172        if self.process_images {
173            self.image_files = WalkDir::new(&self.source_dir)
174                .follow_links(true)
175                .into_iter()
176                .filter_map(|e| e.ok())
177                .filter(|e| e.file_type().is_file())
178                .filter(|e| {
179                    // Filter by image extensions
180                    e.path()
181                        .extension()
182                        .and_then(|ext| ext.to_str())
183                        .map(is_image_file)
184                        .unwrap_or(false)
185                })
186                .map(|e| e.path().to_path_buf())
187                .collect();
188
189            tracing::info!("Found {} image files to process", self.image_files.len());
190        }
191
192        Ok(self)
193    }

195    /// Process all scanned files
196    pub fn process(&mut self) -> Result<&mut Self> {
197        let source_dir = self.source_dir.clone();
198
199        // Process files and collect chunks
200        let results: Vec<_> = self.files
201            .iter()
202            .filter_map(|path| {
203                self.process_file(path, &source_dir).ok()
204            })
205            .collect();
206
207        // Add to chunk store and file map
208        for (entry, chunks) in results {
209            let chunk_refs = self.chunk_store.add_many(chunks);
210
211            // Update manifest with file type info
212            self.manifest.add_file_type(&entry.extension, &entry.path, entry.size);
213
214            // Store file entry with chunk refs
215            let entry_with_refs = FileEntry {
216                chunks: chunk_refs,
217                ..entry
218            };
219            self.file_map.files.insert(entry_with_refs.path.clone(), entry_with_refs);
220        }
221
222        // Update manifest stats
223        let dedup_stats = self.chunk_store.stats();
224        self.manifest.stats.total_files = self.file_map.files.len();
225        self.manifest.stats.unique_chunks = dedup_stats.unique_chunks;
226        self.manifest.stats.original_size_bytes = dedup_stats.total_bytes as u64;
227        self.manifest.stats.dedup_savings_percent = dedup_stats.savings_percent();
228
229        tracing::info!(
230            "Processed {} files, {} unique chunks, {:.1}% dedup savings",
231            self.manifest.stats.total_files,
232            self.manifest.stats.unique_chunks,
233            self.manifest.stats.dedup_savings_percent
234        );
235
236        Ok(self)
237    }

312    /// Generate embeddings for all chunks
313    ///
314    /// This is automatically called during `build()` if embeddings are enabled.
315    /// You can also call it manually after `process()` to inspect the embeddings.
316    #[cfg(all(feature = "embeddings", feature = "search"))]
317    pub fn generate_embeddings(&mut self) -> Result<&mut Self> {
318        let engine = self.embedding_engine.as_ref()
319            .ok_or_else(|| CxpError::Embedding(
320                "Embedding engine not initialized. Call with_embeddings() first.".to_string()
321            ))?;
322
323        tracing::info!("Generating embeddings for {} unique chunks", self.chunk_store.len());
324
325        // Collect all chunk texts
326        let chunks: Vec<_> = self.chunk_store.chunks().collect();
327        let chunk_texts: Vec<&str> = chunks
328            .iter()
329            .map(|c| {
330                std::str::from_utf8(&c.data)
331                    .unwrap_or("[binary data]")
332            })
333            .collect();
334
335        // Process in batches to avoid OOM
336        const BATCH_SIZE: usize = 32;
337        let mut all_embeddings = Vec::new();
338
339        for batch in chunk_texts.chunks(BATCH_SIZE) {
340            let embeddings = engine.embed_batch(batch)?;
341            all_embeddings.extend(embeddings);
342        }
343
344        tracing::info!("Generated {} embeddings", all_embeddings.len());
345
346        // Create quantized embeddings
347        let quantized = QuantizedEmbeddings::from_floats(&all_embeddings);
348
349        tracing::info!(
350            "Quantized embeddings size: {:.2} MB (binary) + {:.2} MB (int8)",
351            quantized.binary.iter().map(|e| e.size_bytes()).sum::<usize>() as f64 / 1024.0 / 1024.0,
352            quantized.int8.iter().map(|e| e.size_bytes()).sum::<usize>() as f64 / 1024.0 / 1024.0
353        );
354
355        // Build HNSW index for binary embeddings
356        let config = HnswConfig::binary(engine.dimensions());
357        let mut index = HnswIndex::new(config)?;
358
359        tracing::info!("Building HNSW index...");
360
361        for (i, binary_emb) in quantized.binary.iter().enumerate() {
362            index.add_binary_embedding(i as u64, binary_emb)?;
363        }
364
365        tracing::info!("HNSW index built with {} vectors", index.len());
366
367        self.chunk_embeddings = Some(quantized);
368        self.search_index = Some(index);
369
370        Ok(self)
371    }

373    /// Add an extension with data to this CXP file
374    ///
375    /// The extension will be registered and its data will be stored in the
376    /// extensions/{namespace}/ directory in the CXP file.
377    ///
378    /// # Arguments
379    /// * `ext` - An object implementing the Extension trait
380    /// * `data` - A HashMap of file names to their data (e.g., "conversations.msgpack" -> bytes)
381    ///
382    /// # Example
383    /// ```ignore
384    /// use std::collections::HashMap;
385    ///
386    /// let mut builder = CxpBuilder::new("./src");
387    /// let contextai = ContextAIExtension::new();
388    /// let data = contextai.to_extension_data()?;
389    ///
390    /// builder.add_extension(&contextai, data)?;
391    /// ```
392    pub fn add_extension<E: Extension + Clone>(
393        &mut self,
394        ext: &E,
395        data: HashMap<String, Vec<u8>>,
396    ) -> Result<&mut Self> {
397        // Register the extension
398        self.extension_manager.register(ext.clone());
399
400        // Add all data files
401        for (key, bytes) in data {
402            self.extension_manager.write_data(ext.namespace(), &key, &bytes)?;
403        }
404
405        // Update manifest to include this extension
406        if !self.manifest.extensions.contains(&ext.namespace().to_string()) {
407            self.manifest.extensions.push(ext.namespace().to_string());
408        }
409
410        tracing::info!(
411            "Added extension '{}' (v{}) with {} data files",
412            ext.namespace(),
413            ext.version(),
414            self.extension_manager.list_data_keys(ext.namespace()).len()
415        );
416
417        Ok(self)
418    }

420    /// Process a single file
421    fn process_file(&self, path: &Path, base_dir: &Path) -> Result<(FileEntry, Vec<Chunk>)> {
422        // Read file content
423        let mut file = File::open(path)?;
424        let metadata = file.metadata()?;
425        let mut content = Vec::with_capacity(metadata.len() as usize);
426        file.read_to_end(&mut content)?;
427
428        // Get relative path
429        let relative_path = path
430            .strip_prefix(base_dir)
431            .unwrap_or(path)
432            .to_string_lossy()
433            .to_string();
434
435        // Get extension
436        let extension = path
437            .extension()
438            .and_then(|e| e.to_str())
439            .unwrap_or("")
440            .to_lowercase();
441
442        // Chunk the content
443        let chunks = chunk_content(&content);
444
445        let entry = FileEntry {
446            path: relative_path,
447            extension,
448            size: metadata.len(),
449            chunks: Vec::new(), // Will be filled in with refs later
450        };
451
452        Ok((entry, chunks))
453    }

455    /// Build and write the CXP file
456    pub fn build<P: AsRef<Path>>(&mut self, output_path: P) -> Result<()> {
457        let output_path = output_path.as_ref();
458        tracing::info!("Building CXP file: {:?}", output_path);
459
460        // Generate embeddings if engine is set but embeddings haven't been generated yet
461        #[cfg(all(feature = "embeddings", feature = "search"))]
462        if self.embedding_engine.is_some() && self.chunk_embeddings.is_none() {
463            self.generate_embeddings()?;
464        }
465
466        let file = File::create(output_path)?;
467        let mut zip = ZipWriter::new(file);
468
469        let options = FileOptions::<()>::default()
470            .compression_method(CompressionMethod::Stored); // We compress chunks ourselves
471
472        // Write manifest
473        let manifest_data = self.manifest.to_msgpack()?;
474        zip.start_file("manifest.msgpack", options.clone())?;
475        zip.write_all(&manifest_data)?;
476
477        // Write file map
478        let file_map_data = rmp_serde::to_vec(&self.file_map)?;
479        zip.start_file("file_map.msgpack", options.clone())?;
480        zip.write_all(&file_map_data)?;
481
482        // Write chunks
483        let chunks: Vec<_> = self.chunk_store.chunks().collect();
484        let total_chunks = chunks.len();
485
486        for (i, chunk) in chunks.iter().enumerate() {
487            let chunk_name = format!("chunks/{}.zst", chunk.id());
488            let compressed = compress(&chunk.data)?;
489
490            zip.start_file(&chunk_name, options.clone())?;
491            zip.write_all(&compressed)?;
492
493            if (i + 1) % 100 == 0 || i + 1 == total_chunks {
494                tracing::debug!("Written {}/{} chunks", i + 1, total_chunks);
495            }
496        }
497
498        // Write embeddings if present
499        #[cfg(all(feature = "embeddings", feature = "search"))]
500        if let Some(ref embeddings) = self.chunk_embeddings {
501            tracing::info!("Writing embeddings to CXP file...");
502
503            // Write binary embeddings
504            let binary_data = serialize_binary_embeddings(&embeddings.binary)?;
505            zip.start_file("embeddings/binary.bin", options.clone())?;
506            zip.write_all(&binary_data)?;
507
508            // Write int8 embeddings
509            let int8_data = serialize_int8_embeddings(&embeddings.int8)?;
510            zip.start_file("embeddings/int8.bin", options.clone())?;
511            zip.write_all(&int8_data)?;
512
513            // Mark that we have embeddings
514            if !self.manifest.extensions.contains(&"embeddings".to_string()) {
515                self.manifest.extensions.push("embeddings".to_string());
516            }
517
518            tracing::info!("Embeddings written successfully");
519        }
520
521        // Write HNSW index if present
522        #[cfg(all(feature = "embeddings", feature = "search"))]
523        if let Some(ref index) = self.search_index {
524            tracing::info!("Writing HNSW index to CXP file...");
525
526            // Save index to a temporary file first (USearch limitation)
527            let temp_dir = std::env::temp_dir();
528            let temp_index_path = temp_dir.join(format!("cxp_index_{}.hnsw", uuid::Uuid::new_v4()));
529
530            index.save(&temp_index_path)?;
531
532            // Read the index file and write to ZIP
533            let mut index_file = File::open(&temp_index_path)?;
534            let mut index_data = Vec::new();
535            index_file.read_to_end(&mut index_data)?;
536
537            zip.start_file("embeddings/index.hnsw", options.clone())?;
538            zip.write_all(&index_data)?;
539
540            // Clean up temp file
541            std::fs::remove_file(&temp_index_path)?;
542
543            tracing::info!("HNSW index written successfully ({} vectors)", index.len());
544        }
545
546        // Write extension data if present
547        if !self.extension_manager.list_extensions().is_empty() {
548            tracing::info!("Writing extension data to CXP file...");
549
550            // Write extension manifests
551            for manifest in self.extension_manager.manifests().values() {
552                let manifest_path = format!("extensions/{}/manifest.msgpack", manifest.namespace);
553                let manifest_data = manifest.to_msgpack()?;
554                zip.start_file(&manifest_path, options.clone())?;
555                zip.write_all(&manifest_data)?;
556            }
557
558            // Write extension data files
559            for (namespace, data_map) in self.extension_manager.all_data() {
560                for (key, data) in data_map {
561                    let data_path = format!("extensions/{}/{}", namespace, key);
562                    zip.start_file(&data_path, options.clone())?;
563                    zip.write_all(data)?;
564                }
565            }
566
567            tracing::info!(
568                "Written {} extensions to CXP file",
569                self.extension_manager.list_extensions().len()
570            );
571        }
572
573        zip.finish()?;
574
575        // Update manifest with final size
576        let final_size = std::fs::metadata(output_path)?.len();
577        self.manifest.stats.cxp_size_bytes = final_size;
578        self.manifest.stats.compression_ratio =
579            final_size as f64 / self.manifest.stats.original_size_bytes as f64;
580
581        tracing::info!(
582            "CXP file created: {:.2} MB (compression ratio: {:.2}%)",
583            final_size as f64 / 1024.0 / 1024.0,
584            self.manifest.stats.compression_ratio * 100.0
585        );
586
587        Ok(())
588    }

595    /// File map
596    pub file_map: FileMap,
597    /// ZIP archive handle
598    archive_path: PathBuf,
599    /// Extension manager for reading app-specific data
600    extension_manager: ExtensionManager,
601    /// Cached HNSW index for semantic search
602    #[cfg(all(feature = "embeddings", feature = "search"))]
603    search_index: Option<HnswIndex>,
604    /// Cached embeddings for rescoring
605    #[cfg(all(feature = "embeddings", feature = "search"))]

608
609impl CxpReader {
610    /// Open a CXP file for reading
611    pub fn open<P: AsRef<Path>>(path: P) -> Result<Self> {
612        let path = path.as_ref().to_path_buf();
613        let file = File::open(&path)?;
614        let mut archive = ZipArchive::new(file)?;
615
616        // Read manifest
617        let manifest = {
618            let mut manifest_file = archive.by_name("manifest.msgpack")?;
619            let mut data = Vec::new();
620            manifest_file.read_to_end(&mut data)?;
621            Manifest::from_msgpack(&data)?
622        };
623
624        // Read file map
625        let file_map = {
626            let mut file_map_file = archive.by_name("file_map.msgpack")?;
627            let mut data = Vec::new();
628            file_map_file.read_to_end(&mut data)?;
629            rmp_serde::from_slice(&data)?
630        };
631
632        // Load extension data if present
633        let mut extension_manager = ExtensionManager::new();
634
635        // Iterate through all files in the ZIP archive to find extensions
636        for i in 0..archive.len() {
637            let file = archive.by_index(i)?;
638            let file_name = file.name().to_string();
639
640            // Check if this is an extension file
641            if file_name.starts_with("extensions/") {
642                let parts: Vec<&str> = file_name.split('/').collect();
643                if parts.len() >= 3 {
644                    let namespace = parts[1];
645                    let file_key = parts[2..].join("/");
646
647                    // Read the file data
648                    let mut data = Vec::new();
649                    drop(file); // Close the borrowed file
650                    let mut file = archive.by_index(i)?;
651                    file.read_to_end(&mut data)?;
652
653                    if file_key == "manifest.msgpack" {
654                        // Load extension manifest
655                        if let Ok(ext_manifest) = crate::extensions::ExtensionManifest::from_msgpack(&data) {
656                            extension_manager.load_manifest(ext_manifest);
657                        }
658                    } else {
659                        // Load extension data
660                        extension_manager.load_data(namespace.to_string(), file_key, data);
661                    }
662                }
663            }
664        }
665
666        if !extension_manager.list_extensions().is_empty() {
667            tracing::info!(
668                "Loaded {} extensions from CXP file",
669                extension_manager.list_extensions().len()
670            );
671        }
672
673        Ok(Self {
674            manifest,
675            file_map,
676            archive_path: path,
677            extension_manager,
678            #[cfg(all(feature = "embeddings", feature = "search"))]
679            search_index: None,
680            #[cfg(all(feature = "embeddings", feature = "search"))]
681            embeddings: None,
682        })
683    }

695    /// Read a file's content by reconstructing from chunks
696    pub fn read_file(&self, path: &str) -> Result<Vec<u8>> {
697        let entry = self.file_map.files.get(path)
698            .ok_or_else(|| CxpError::FileNotFound(path.to_string()))?;
699
700        let file = File::open(&self.archive_path)?;
701        let mut archive = ZipArchive::new(file)?;
702
703        let mut content = Vec::with_capacity(entry.size as usize);
704
705        for chunk_ref in &entry.chunks {
706            let chunk_name = format!("chunks/{}.zst", &chunk_ref.hash[..16]);
707            let mut chunk_file = archive.by_name(&chunk_name)?;
708
709            let mut compressed = Vec::new();
710            chunk_file.read_to_end(&mut compressed)?;
711
712            let decompressed = decompress(&compressed)?;
713            content.extend_from_slice(&decompressed);
714        }
715
716        Ok(content)
717    }

733    ///
734    /// Returns the embeddings without loading the HNSW index or caching them.
735    /// Use this if you only need to access the embeddings directly.
736    #[cfg(any(feature = "embeddings", feature = "embeddings-wasm"))]

738        if !self.has_embeddings() {
739            return Err(CxpError::Embedding(
740                "This CXP file does not contain embeddings".to_string()
741            ));
742        }
743
744        tracing::info!("Loading embeddings from CXP file...");
745
746        let file = File::open(&self.archive_path)?;
747        let mut archive = ZipArchive::new(file)?;
748
749        // Load binary embeddings
750        let binary_embeddings = {
751            let mut file = archive.by_name("embeddings/binary.bin")?;
752            let mut data = Vec::new();
753            file.read_to_end(&mut data)?;
754            deserialize_binary_embeddings(&data)?
755        };
756
757        // Load int8 embeddings
758        let int8_embeddings = {
759            let mut file = archive.by_name("embeddings/int8.bin")?;
760            let mut data = Vec::new();
761            file.read_to_end(&mut data)?;
762            deserialize_int8_embeddings(&data)?
763        };
764
765        let dimensions = self.manifest.embedding_dim
766            .ok_or_else(|| CxpError::Embedding("No embedding dimension in manifest".to_string()))?;
767
768        tracing::info!("Loaded {} embeddings", binary_embeddings.len());

770        Ok(crate::EmbeddingStore::new(
771            binary_embeddings,
772            int8_embeddings,
773            dimensions,
774        ))

777    /// Load embeddings and search index into memory
778    ///
779    /// This must be called before using semantic search functions.
780    /// The embeddings and index are cached for subsequent searches.
781    #[cfg(all(feature = "embeddings", feature = "search"))]
782    pub fn load_embeddings(&mut self) -> Result<()> {
783        if !self.has_embeddings() {
784            return Err(CxpError::Embedding(
785                "This CXP file does not contain embeddings".to_string()
786            ));
787        }
788
789        if self.search_index.is_some() {
790            return Ok(());  // Already loaded
791        }
792
793        tracing::info!("Loading embeddings from CXP file...");
794
795        let file = File::open(&self.archive_path)?;
796        let mut archive = ZipArchive::new(file)?;
797
798        // Load binary embeddings
799        let binary_embeddings = {
800            let mut file = archive.by_name("embeddings/binary.bin")?;
801            let mut data = Vec::new();
802            file.read_to_end(&mut data)?;
803            deserialize_binary_embeddings(&data)?
804        };
805
806        // Load int8 embeddings
807        let int8_embeddings = {
808            let mut file = archive.by_name("embeddings/int8.bin")?;
809            let mut data = Vec::new();
810            file.read_to_end(&mut data)?;
811            deserialize_int8_embeddings(&data)?
812        };
813
814        tracing::info!("Loaded {} embeddings", binary_embeddings.len());
815
816        self.embeddings = Some(QuantizedEmbeddings {
817            binary: binary_embeddings,
818            int8: int8_embeddings,
819        });
820
821        // Load HNSW index
822        let file = File::open(&self.archive_path)?;
823        let mut archive = ZipArchive::new(file)?;
824
825        let mut index_file = archive.by_name("embeddings/index.hnsw")?;
826        let mut index_data = Vec::new();
827        index_file.read_to_end(&mut index_data)?;
828
829        // Save to temp file (USearch limitation)
830        let temp_dir = std::env::temp_dir();
831        let temp_index_path = temp_dir.join(format!("cxp_index_{}.hnsw", uuid::Uuid::new_v4()));
832
833        let mut temp_file = File::create(&temp_index_path)?;
834        temp_file.write_all(&index_data)?;
835        drop(temp_file);
836
837        // Load index
838        let dimensions = self.manifest.embedding_dim
839            .ok_or_else(|| CxpError::Embedding("No embedding dimension in manifest".to_string()))?;
840
841        let config = HnswConfig::binary(dimensions);
842        let index = HnswIndex::load(&temp_index_path, config)?;
843
844        // Clean up temp file
845        std::fs::remove_file(&temp_index_path)?;
846
847        tracing::info!("Loaded HNSW index with {} vectors", index.len());
848
849        self.search_index = Some(index);
850
851        Ok(())
852    }

854    /// Perform semantic search using a query embedding
855    ///
856    /// Returns the top-k most similar chunks by ID.
857    /// You must call `load_embeddings()` first.
858    ///
859    /// # Arguments
860    /// * `query_embedding` - The query vector (should match the model's dimensions)
861    /// * `top_k` - Number of results to return
862    ///
863    /// # Returns
864    /// Vector of (chunk_id, similarity_score) tuples, sorted by similarity (highest first)
865    #[cfg(all(feature = "embeddings", feature = "search"))]
866    pub fn search_semantic(
867        &self,
868        query_embedding: &[f32],
869        top_k: usize,
870    ) -> Result<Vec<SearchResult>> {
871        let index = self.search_index.as_ref()
872            .ok_or_else(|| CxpError::Search(
873                "Embeddings not loaded. Call load_embeddings() first.".to_string()
874            ))?;
875
876        let embeddings = self.embeddings.as_ref()
877            .ok_or_else(|| CxpError::Search(
878                "Embeddings not loaded. Call load_embeddings() first.".to_string()
879            ))?;
880
881        // Convert query to binary for fast initial search
882        let query_binary = BinaryEmbedding::from_float(query_embedding);
883
884        // Search with HNSW (binary)
885        let candidates = index.search_binary_embedding(&query_binary, top_k * 2)?;
886
887        // Rescore with Int8 for better accuracy
888        let query_int8 = Int8Embedding::from_float(query_embedding);
889
890        let mut rescored: Vec<_> = candidates
891            .iter()
892            .map(|result| {
893                let chunk_id = result.id as usize;
894                let score = if chunk_id < embeddings.int8.len() {
895                    embeddings.int8[chunk_id].dot_product(&query_int8)
896                } else {
897                    0.0
898                };
899                SearchResult {
900                    id: result.id,
901                    distance: -score,  // Negate for sorting (higher is better)
902                }
903            })
904            .collect();
905
906        // Sort by score (descending)
907        rescored.sort_by(|a, b| a.distance.partial_cmp(&b.distance).unwrap());
908
909        // Take top-k and fix the distance sign
910        Ok(rescored
911            .into_iter()
912            .take(top_k)
913            .map(|mut r| {
914                r.distance = -r.distance;
915                r
916            })
917            .collect())
918    }

920    /// Get chunk text by ID
921    ///
922    /// This is useful for retrieving the actual content of chunks found by semantic search.
923    #[cfg(all(feature = "embeddings", feature = "search"))]
924    pub fn get_chunk_text(&self, chunk_id: u64) -> Result<String> {
925        let file = File::open(&self.archive_path)?;
926        let mut archive = ZipArchive::new(file)?;
927
928        // We need to find the chunk by iterating through the chunk store
929        // For now, we'll use the chunk hash format
930        let chunk_name = format!("chunks/{:016x}.zst", chunk_id);
931
932        let mut chunk_file = archive.by_name(&chunk_name)
933            .or_else(|_| {
934                // Try alternative naming if the first format doesn't work
935                // This is a fallback - in practice you'd maintain a chunk ID -> hash mapping
936                Err(CxpError::FileNotFound(format!("Chunk {} not found", chunk_id)))
937            })?;
938
939        let mut compressed = Vec::new();
940        chunk_file.read_to_end(&mut compressed)?;
941
942        let decompressed = decompress(&compressed)?;
943
944        String::from_utf8(decompressed)
945            .map_err(|e| CxpError::Serialization(format!("Invalid UTF-8 in chunk: {}", e)))
946    }
